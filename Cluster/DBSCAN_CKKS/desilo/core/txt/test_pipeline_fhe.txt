from desilofhe import Engine

from core.ciphertext.client.MultipartyDataOwner import prepare_and_encrypt_owner_blocks

def load_arff_to_pts_with_labels(filepath: str) -> Tuple[np.ndarray, np.ndarray]:
    pts, true_labels = [], []
    data_section = False
    with open(filepath, "r", encoding="utf-8") as f:
        for line in f:
            line = line.strip()
            if not line or line.startswith("%"):
                continue
            if line.lower().startswith("@data"):
                data_section = True
                continue
            if data_section:
                line = line.replace("\t", " ").replace(",", " ")
                values = line.split()
                if len(values) < 2:
                    continue
                pts.append([float(v) for v in values[:-1]])
                true_labels.append(int(float(values[-1])))
    if not pts:
        raise ValueError("데이터를 찾을 수 없습니다.")
    return np.array(pts, dtype=np.float64), np.array(true_labels, dtype=int)


def remap_labels_to_sequential(labels: List[int]) -> List[int]:
    unique_valid = sorted(x for x in set(labels) if x != -1)
    mapping = {old: idx + 1 for idx, old in enumerate(unique_valid)}
    return [-1 if x == -1 else mapping[x] for x in labels]


def save_timings_txt(filename: str, timings_dict: dict) -> None:
    try:
        with open(filename, "w", encoding="utf-8") as f:
            for k, v in timings_dict.items():
                f.write(f"{k}: {float(v):.6f}\n")
        print(f"✅ 타이밍 파일 저장 완료: {filename}")
    except Exception as e:
        print(f"❌ 타이밍 파일 저장 실패: {e}")


# ─────────────────────────────────────────────────────────────────
# 메인
# ─────────────────────────────────────────────────────────────────

def main():
    print("=" * 60)
    print("  FHE Multiparty DBSCAN 테스트 (sign_bootstrap_key 전용)")
    print("=" * 60)
    print(f"\n▶ 데이터셋: {DATASET_PATH}\n")

    eps_val     = float(input("eps 값을 입력하세요 (예: 0.5) > "))
    min_pts_val = int(float(input("min_pts 값을 입력하세요 (예: 3) > ")))
    n_dos       = int(input("DO 수를 입력하세요 (예: 1) > "))

    print(f"\n  eps={eps_val}, min_pts={min_pts_val}, n_dos={n_dos}")

    # ── 데이터 로딩 ───────────────────────────────────────────
    print("\n데이터 로딩 중...")
    pts, true_labels = load_arff_to_pts_with_labels(DATASET_PATH)
    N, DIM = pts.shape
    print(f"  완료: {N}개 점, {DIM}차원")

    global_min   = float(np.min(pts))
    global_max   = float(np.max(pts))
    scale_factor = global_max - global_min if (global_max - global_min) != 0.0 else 1.0

    query_epsilon_norm = eps_val / scale_factor
    base_epsilon_norm  = min((min_pts_val / N) ** (1.0 / DIM), 1.0)
    domain_mins_norm   = [0.0] * DIM
    domain_maxs_norm   = [1.0] * DIM

    print(f"  global_min={global_min:.4f}, global_max={global_max:.4f}")
    print(f"  query_eps_norm={query_epsilon_norm:.6f}, "
          f"base_eps_norm={base_epsilon_norm:.6f}")

    # ── Grid 준비 ──────────────────────────────────────────────
    print("\n" + "=" * 60 + "\n  Grid 준비\n" + "=" * 60)
    grid_centers_norm = generate_public_grid_centers_nd(
        domain_mins=domain_mins_norm,
        domain_maxs=domain_maxs_norm,
        epsilon=base_epsilon_norm,
    )
    axis_cell_counts = compute_axis_cell_counts(
        domain_mins=domain_mins_norm,
        domain_maxs=domain_maxs_norm,
        epsilon=base_epsilon_norm,
    )
    num_grids  = 1
    for c in axis_cell_counts: num_grids *= c
    bucket_size = 32
    N_batch     = bucket_size * num_grids
    print(f"  축별 cell 수: {axis_cell_counts}, G={num_grids}, "
          f"bucket_size={bucket_size}, N_batch={N_batch}")

    if len(grid_centers_norm) != num_grids:
        raise ValueError("grid center 개수 불일치")

    # ── FHE 엔진 / KeyPack 생성 ───────────────────────────────
    # bootstrap_key 완전 미사용: sign_bootstrap_key(LossyBootstrapKey)만 생성
    print("\n" + "=" * 60 + "\n  FHE 엔진 / 키 생성 (sign_bootstrap_key 전용)\n" + "=" * 60)
    engine     = Engine(use_bootstrap=True, mode="gpu")
    secret_key = engine.create_secret_key()

    keypack = KeyPack(
        public_key          = engine.create_public_key(secret_key),
        rotation_key        = engine.create_rotation_key(secret_key),
        relinearization_key = engine.create_relinearization_key(secret_key),
        conjugation_key     = engine.create_conjugation_key(secret_key),
        sign_bootstrap_key  = engine.create_lossy_bootstrap_key(secret_key),
        # bootstrap_key 완전 제거 — create_bootstrap_key() 호출 없음
    )
    print("  ✅ KeyPack 생성 완료 (sign_bootstrap_key only)")

    # ── DO 데이터 분배 ────────────────────────────────────────
    chunk      = N // n_dos
    splits_raw = [
        pts[i * chunk : (i + 1) * chunk if i < n_dos - 1 else N].tolist()
        for i in range(n_dos)
    ]
    true_labels_split = [
        true_labels[i * chunk : (i + 1) * chunk if i < n_dos - 1 else N].tolist()
        for i in range(n_dos)
    ]

    # ── DO 암호화 ─────────────────────────────────────────────
    print("\n" + "=" * 60 + "\n  Data Owner 암호화\n" + "=" * 60)
    owner_client_packs    = []
    encrypted_owner_packs = []

    fhe_start = time()
    for owner_id in range(n_dos):
        print(f"  DO_{owner_id}: {len(splits_raw[owner_id])}개 점 암호화 중...")
        c_pack, s_pack = prepare_and_encrypt_owner_blocks(
            engine            = engine,
            keypack           = keypack,
            owner_points_raw  = splits_raw[owner_id],
            grid_centers_norm = grid_centers_norm,
            epsilon_norm      = base_epsilon_norm,
            bucket_size       = bucket_size,
            global_min        = global_min,
            global_max        = global_max,
            owner_id          = owner_id,
        )
        owner_client_packs.append(c_pack)
        encrypted_owner_packs.append(s_pack)

    plain_all_points, total_points = assign_global_indices(owner_client_packs)
    owner_coord_map = build_owner_coord_map(owner_client_packs)
    print(f"  총 유효 포인트: {total_points}")

    # ── Server DBSCAN ─────────────────────────────────────────
    print("\n" + "=" * 60 + "\n  Server FHE-DBSCAN\n" + "=" * 60)
    t_server = time()
    final_ct = run_multiparty_point_dbscan(
        engine                   = engine,
        keypack                  = keypack,
        encrypted_owner_packs    = encrypted_owner_packs,
        grid_centers_norm        = grid_centers_norm,
        query_epsilon_norm       = query_epsilon_norm,
        base_epsilon_norm        = base_epsilon_norm,
        min_pts                  = min_pts_val,
        bucket_size              = bucket_size,
        total_points_upper_bound = N_batch,
    )
    t_server_elapsed = time() - t_server

    # ── 복호화 ───────────────────────────────────────────────
    print("\n" + "=" * 60 + "\n  복호화 & 라벨 추출\n" + "=" * 60)
    t_decrypt = time()
    results = reconstruct_results(
        engine         = engine,
        secret_key     = secret_key,
        final_ct       = final_ct,
        plain_all_points = plain_all_points,
        owner_coord_map  = owner_coord_map,
        global_min       = global_min,
        scale_factor     = scale_factor,
    )
    ordered_results     = sorted(results, key=lambda x: (x["owner_id"], x["owner_local_idx"]))
    cluster_labels_fhe  = [int(row["label"]) for row in ordered_results]
    cluster_labels_fhe  = remap_labels_to_sequential(cluster_labels_fhe)
    t_decrypt_elapsed   = time() - t_decrypt
    t_total_elapsed     = time() - fhe_start

    # ── 결과 샘플 ─────────────────────────────────────────────
    print("\n결과 샘플 (상위 20개):")
    for row in ordered_results[:min(20, len(ordered_results))]:
        idx = ordered_results.index(row)
        print(
            f"  gid={row['global_idx']:4d} | owner={row['owner_id']} | "
            f"local={row['owner_local_idx']:4d} | "
            f"pred={cluster_labels_fhe[idx]:3d} | "
            f"true={true_labels[idx]}"
        )

    # ── true label 직접 비교 ──────────────────────────────────
    print("\n" + "=" * 60 + "\n  True Label vs FHE DBSCAN 직접 비교\n" + "=" * 60)

    # DO 분배 순서대로 전체 true label 재구성
    all_true_labels: List[int] = []
    for i in range(n_dos):
        all_true_labels.extend(true_labels_split[i])
    true_labels_seq = remap_labels_to_sequential(all_true_labels)

    n_clusters_pred = len([c for c in set(cluster_labels_fhe) if c != -1])
    n_clusters_true = len([c for c in set(true_labels_seq)    if c != -1])
    n_noise_pred    = cluster_labels_fhe.count(-1)
    n_noise_true    = true_labels_seq.count(-1)

    print(f"  예측 클러스터 수: {n_clusters_pred}  (noise: {n_noise_pred}/{N})")
    print(f"  정답 클러스터 수: {n_clusters_true}  (noise: {n_noise_true}/{N})")

    # 점별 직접 비교 (상위 50개)
    print(f"\n  [점별 직접 비교 — 상위 50개]")
    print(f"  {'idx':>5} | {'pred':>5} | {'true':>5} | {'match':>5}")
    print("  " + "-" * 35)
    n_match = 0
    for i in range(min(50, len(cluster_labels_fhe))):
        match = "✓" if cluster_labels_fhe[i] == true_labels_seq[i] else "✗"
        if cluster_labels_fhe[i] == true_labels_seq[i]:
            n_match += 1
        print(f"  {i:>5} | {cluster_labels_fhe[i]:>5} | "
              f"{true_labels_seq[i]:>5} | {match:>5}")
    top50_match_pct = n_match / min(50, len(cluster_labels_fhe)) * 100
    print(f"\n  상위 50개 정확 일치율: {top50_match_pct:.1f}%")
    print("  ⚠️  라벨 ID는 임의 순열이므로 ARI 지표가 실제 성능 기준입니다.")

    # ARI
    ari_score = adjusted_rand_score(true_labels_seq, cluster_labels_fhe)
    print(f"\n📊 ARI: {ari_score * 100:.2f} / 100점")
    if ari_score > 0.99:
        print("  => 🎉 대성공!")
    elif ari_score > 0.80:
        print("  => 👍 대부분 동일, 경계 일부 차이")
    else:
        print("  => ❌ eps/min_pts 재점검 필요")

    # ── 타이밍 ───────────────────────────────────────────────
    timings = {
        "encrypt_sec":  fhe_start - fhe_start,   # 암호화 포함하려면 별도 측정
        "server_sec":   t_server_elapsed,
        "decrypt_sec":  t_decrypt_elapsed,
        "total_sec":    t_total_elapsed,
    }
    print(f"\n타이밍:")
    for k, v in timings.items():
        print(f"  {k:<20s}: {v:.4f}초")
    save_timings_txt(
        f"timings_fhe_eps{eps_val}_min{min_pts_val}_dos{n_dos}.txt", timings
    )

    # ── CSV 저장 ─────────────────────────────────────────────
    out_csv = f"result_fhe_eps{eps_val}_min{min_pts_val}_dos{n_dos}.csv"
    try:
        with open(out_csv, "w", encoding="utf-8") as f:
            axis_headers = ",".join([f"x{i+1}" for i in range(DIM)])
            f.write(f"{axis_headers},Pred_Cluster,True_Class\n")
            for i, row in enumerate(ordered_results):
                pt = row.get("point_reconstructed") or []
                coords = ",".join([f"{v:.6f}" for v in pt]) if pt else ",".join(["NA"] * DIM)
                f.write(f"{coords},{cluster_labels_fhe[i]},{true_labels_seq[i]}\n")
        print(f"✅ 결과 CSV 저장: {out_csv}")
    except Exception as e:
        print(f"❌ CSV 저장 실패: {e}")


if __name__ == "__main__":
    main()
