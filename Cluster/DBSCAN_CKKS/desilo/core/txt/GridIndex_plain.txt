from typing import List
import math
import itertools


def normalize_points_global(points, global_min, global_max):
    scale = global_max - global_min
    if scale == 0.0:
        scale = 1.0

    out = []
    for row in points:
        out.append([(v - global_min) / scale for v in row])
    return out, scale


def normalize_grid_centers(grid_centers, global_min, global_max):
    scale = global_max - global_min
    if scale == 0.0:
        scale = 1.0

    out = []
    for c in grid_centers:
        out.append([(v - global_min) / scale for v in c])
    return out, scale


def generate_public_grid_centers_nd(domain_mins, domain_maxs, epsilon):
    """
    n차원 public grid center 생성.
    각 축마다 [min + eps/2, min + 3eps/2, ...] 방식으로 center를 생성.
    """
    axis_centers = []

    for mn, mx in zip(domain_mins, domain_maxs):
        centers_1d = []
        x = mn + epsilon / 2.0
        while x < mx:
            centers_1d.append(x)
            x += epsilon

        if len(centers_1d) == 0:
            centers_1d = [mn + epsilon / 2.0]

        axis_centers.append(centers_1d)

    grid_centers = [list(p) for p in itertools.product(*axis_centers)]
    return grid_centers


def build_grid_adjacency(grid_centers_norm, epsilon_norm):
    num_grids = len(grid_centers_norm)
    adjacency = [[0] * num_grids for _ in range(num_grids)]
    eps_sq = epsilon_norm ** 2

    for i in range(num_grids):
        for j in range(num_grids):
            if i == j:
                adjacency[i][j] = 1
            else:
                dist_sq = sum((a - b) ** 2 for a, b in zip(grid_centers_norm[i], grid_centers_norm[j]))
                if dist_sq <= eps_sq:
                    adjacency[i][j] = 1
    return adjacency


def compute_axis_cell_counts(domain_mins_norm, domain_maxs_norm, epsilon_norm):
    counts = []
    for mn, mx in zip(domain_mins_norm, domain_maxs_norm):
        c = int(round((mx - mn) / epsilon_norm))
        if c <= 0:
            c = 1
        counts.append(c)
    return counts


def flatten_multi_index(indices, axis_cell_counts):
    """
    [i0, i1, i2, ...] -> 1차 index
    row-major 방식.
    """
    flat = 0
    stride = 1
    for axis in range(len(indices) - 1, -1, -1):
        flat += indices[axis] * stride
        stride *= axis_cell_counts[axis]
    return flat


def point_to_grid_index_direct_nd(point_norm,
                                  domain_mins_norm,
                                  domain_maxs_norm,
                                  epsilon_norm,
                                  axis_cell_counts):
    if len(point_norm) != len(domain_mins_norm):
        return None

    indices = []
    dim = len(point_norm)

    for d in range(dim):
        x = point_norm[d]
        mn = domain_mins_norm[d]
        mx = domain_maxs_norm[d]

        if not (mn <= x <= mx):
            return None

        if x == mx:
            g = axis_cell_counts[d] - 1
        else:
            g = int((x - mn) // epsilon_norm)

        if g < 0:
            g = 0
        if g >= axis_cell_counts[d]:
            g = axis_cell_counts[d] - 1

        indices.append(g)

    return flatten_multi_index(indices, axis_cell_counts)


def bucketize_points_by_grid(points_norm,
                             points_raw,
                             domain_mins_norm,
                             domain_maxs_norm,
                             epsilon_norm,
                             axis_cell_counts,
                             bucket_size,
                             max_blocks_per_grid,
                             owner_id):
    dim = len(points_norm[0]) if points_norm else 0

    num_grids = 1
    for c in axis_cell_counts:
        num_grids *= c

    grid_to_points = {g: [] for g in range(num_grids)}

    for local_idx, (p_norm, p_raw) in enumerate(zip(points_norm, points_raw)):
        g = point_to_grid_index_direct_nd(
            point_norm=p_norm,
            domain_mins_norm=domain_mins_norm,
            domain_maxs_norm=domain_maxs_norm,
            epsilon_norm=epsilon_norm,
            axis_cell_counts=axis_cell_counts
        )
        if g is not None:
            grid_to_points[g].append({
                "point_norm": p_norm,
                "point_raw": p_raw,
                "owner_id": owner_id,
                "owner_local_idx": local_idx
            })

    for g in range(num_grids):
        real_count = len(grid_to_points[g])
        if real_count > 0:
            print(f"[DEBUG] owner={owner_id}, grid={g}, real_count={real_count}")

    blocks = []
    for g in range(num_grids):
        pts = grid_to_points[g]
        needed_blocks = math.ceil(len(pts) / bucket_size) if pts else 1
        needed_blocks = min(needed_blocks, max_blocks_per_grid)

        cursor = 0
        for b in range(max_blocks_per_grid):
            if b < needed_blocks:
                sub = pts[cursor:cursor + bucket_size]
                cursor += bucket_size
            else:
                sub = []

            padded_norm = []
            padded_raw = []
            selection_mask = []
            point_refs = []

            for item in sub:
                padded_norm.append(item["point_norm"])
                padded_raw.append(item["point_raw"])
                selection_mask.append(1)
                point_refs.append({
                    "owner_id": item["owner_id"],
                    "owner_local_idx": item["owner_local_idx"],
                    "global_idx": None
                })

            while len(padded_norm) < bucket_size:
                padded_norm.append([0.0] * dim)
                padded_raw.append(None)
                selection_mask.append(0)
                point_refs.append(None)

            blocks.append({
                "grid_idx": g,
                "block_idx": b,
                "points_norm": padded_norm,
                "points_raw": padded_raw,
                "selection_mask": selection_mask,
                "point_refs": point_refs
            })

    return blocks
