from desilofhe import Engine, Ciphertext
from util.keypack import KeyPack
from core.ciphertext_single.minimax import load_mcp


# ── α=8 MCP 계수: 모듈 로드 시 1회만 읽음 ───────────────────────────────────
# mcp_path 는 test 실행 전 자동 생성됨 (test_Server_main_call_dataset.py 참조)
_MCP_ALPHA8_PATH = "mcp_alpha8.json"
_mcp_alpha8_components = None   # lazy load


def _get_mcp_alpha8():
    global _mcp_alpha8_components
    if _mcp_alpha8_components is None:
        print(f"  [LabelProp] α=8 MCP 계수 로드 ({_MCP_ALPHA8_PATH})")
        _mcp_alpha8_components = load_mcp(_MCP_ALPHA8_PATH)
    return _mcp_alpha8_components


def _refresh(engine: Engine, ct: Ciphertext, keypack: KeyPack) -> Ciphertext:
    return engine.bootstrap(
        engine.intt(ct),
        keypack.relinearization_key,
        keypack.conjugation_key,
        keypack.bootstrap_key
    )


def _eval_mcp_fhe(engine, ct, components, slot_count, keypack):
    """
    FHE 상에서 Minimax Composite Polynomial 평가.
    중간 단계 bootstrap 포함 (마지막 제외).
    """
    relin_key = keypack.relinearization_key
    conj_key  = keypack.conjugation_key
    boot_key  = keypack.bootstrap_key
    current   = ct

    for step_idx, comp in enumerate(components):
        coeffs = comp["coeffs"]
        x_sq   = engine.square(current, relin_key)
        x_pow  = current
        result = engine.multiply(
            x_pow, engine.encode([coeffs[0]] * slot_count)
        )
        for k in range(1, len(coeffs)):
            x_pow  = engine.multiply(x_pow, x_sq, relin_key)
            result = engine.add(
                result,
                engine.multiply(x_pow, engine.encode([coeffs[k]] * slot_count))
            )
        current = result

        if step_idx < len(components) - 1:
            current = engine.intt(current)
            current = engine.bootstrap(current, relin_key, conj_key, boot_key)

    return current


def fhe_circular_shift(engine, ct, k, num_points, keypack):
    """
    암호화된 벡터를 k 칸 왼쪽으로 circular shift.
    np.roll(ct, -k) 의 FHE 구현.
    """
    slot_count = engine.slot_count

    left_shifted  = engine.rotate(ct, keypack.rotation_key, -k)
    right_shifted = engine.rotate(ct, keypack.rotation_key, num_points - k)

    mask_left = [1.0 if i < (num_points - k) else 0.0 for i in range(slot_count)]
    mask_right = [0.0] * slot_count
    for i in range(num_points - k, num_points):
        mask_right[i] = 1.0

    left_clean  = engine.multiply(left_shifted,  engine.encode(mask_left))
    right_clean = engine.multiply(right_shifted, engine.encode(mask_right))
    return engine.add(left_clean, right_clean)


def fhe_sign_unit(
    engine     : Engine,
    x_ct       : Ciphertext,
    num_points : int,
    keypack    : KeyPack,
    depth      : int = 5,              # 하위 호환성 유지 (내부 미사용)
    bootstrap_interval: int = 3,       # 하위 호환성 유지 (내부 미사용)
) -> Ciphertext:
    """
    α=8 MCP 로 sign(x) 근사.

    기존 cubic 반복 (depth=5) 에서 mcp_alpha8.json 로드 방식으로 교체.
    depth / bootstrap_interval 파라미터는 하위 호환성을 위해 시그니처 유지,
    내부에서는 MCP 계수를 사용함.

    α=8 설계 근거 (Label Propagation):
      - 매 fhe_fast_max_unit 호출 후 _refresh(bootstrap) 실행
        → 오차가 iteration 간 누적되지 않고 리셋
      - cluster_id gap = 1/N, sign_err(α=8) ≈ 3.7e-3
        → fhe_fast_max_unit_scaled 에서 diff/id_gap 으로 정규화하여 구별 보장
    """
    slot_count = engine.slot_count
    components = _get_mcp_alpha8()
    return _eval_mcp_fhe(engine, x_ct, components, slot_count, keypack)


def fhe_fast_max_unit(
    engine     : Engine,
    A_ct       : Ciphertext,
    B_ct       : Ciphertext,
    num_points : int,
    keypack    : KeyPack,
    depth      : int = 5,
) -> Ciphertext:
    """max(A, B) = (A + B + (B-A)·sign(B-A)) / 2"""
    relin_key = keypack.relinearization_key
    slot_count = engine.slot_count
    half_pt    = engine.encode([0.5] * slot_count)

    diff_ct      = engine.subtract(B_ct, A_ct)
    sign_ct      = fhe_sign_unit(engine, diff_ct, num_points, keypack)
    sign_ct      = _refresh(engine, sign_ct, keypack)
    diff_ct_ref  = _refresh(engine, diff_ct, keypack)
    diff_sign_ct = engine.multiply(diff_ct_ref, sign_ct, relin_key)
    diff_sign_ct = _refresh(engine, diff_sign_ct, keypack)
    relu_ct      = engine.multiply(engine.add(diff_ct_ref, diff_sign_ct), half_pt)
    return _refresh(engine, engine.add(A_ct, relu_ct), keypack)


def fhe_hard_mask01(
    engine     : Engine,
    x_ct       : Ciphertext,
    num_points : int,
    keypack    : KeyPack,
    depth      : int = 5,
) -> Ciphertext:
    """{0,1} 마스킹: sign(x - 0.5) 기반."""
    slot_count = engine.slot_count
    half_pt    = engine.encode([0.5] * slot_count)
    centered   = engine.subtract(x_ct, half_pt)
    centered   = _refresh(engine, centered, keypack)
    sign_ct    = fhe_sign_unit(engine, centered, num_points, keypack)
    out_ct     = engine.add(engine.multiply(sign_ct, half_pt), half_pt)
    return _refresh(engine, out_ct, keypack)


def fhe_fast_max_unit_scaled(
    engine     : Engine,
    A_ct       : Ciphertext,
    B_ct       : Ciphertext,
    num_points : int,
    keypack    : KeyPack,
    depth      : int = 5,
    id_gap     : float = None,
) -> Ciphertext:
    """
    cluster_id gap 을 고려한 max 연산.
    diff / id_gap 으로 정규화 후 sign → 구별 보장.
    """
    relin_key  = keypack.relinearization_key
    slot_count = engine.slot_count
    half_pt    = engine.encode([0.5] * slot_count)

    diff_ct = engine.subtract(B_ct, A_ct)

    if id_gap is not None and id_gap > 0:
        scale    = 1.0 / (id_gap * num_points)
        diff_sc  = engine.multiply(diff_ct, engine.encode([scale] * slot_count))
        diff_sc  = _refresh(engine, diff_sc, keypack)
    else:
        diff_sc  = diff_ct

    sign_ct      = fhe_sign_unit(engine, diff_sc, num_points, keypack)
    sign_ct      = _refresh(engine, sign_ct, keypack)
    diff_ct_ref  = _refresh(engine, diff_ct, keypack)
    diff_sign_ct = engine.multiply(diff_ct_ref, sign_ct, relin_key)
    diff_sign_ct = _refresh(engine, diff_sign_ct, keypack)
    relu_ct      = engine.multiply(engine.add(diff_ct_ref, diff_sign_ct), half_pt)
    return _refresh(engine, engine.add(A_ct, relu_ct), keypack)


def fhe_max_propagation_fhe(
    engine          : Engine,
    keypack         : KeyPack,
    adjacency_ct_list: list,
    core_ct         : Ciphertext,
    cluster_id_pt   : list,
    num_points      : int,
    max_iter        : int = None,
) -> Ciphertext:
    """
    FHE DBSCAN Label Propagation.
    (로직 완전 동일 — sign 근사만 α=8 MCP 로 교체)
    """
    relin_key          = keypack.relinearization_key
    id_gap             = 1.0 / num_points
    cluster_id_encoded = engine.encode(cluster_id_pt)

    if max_iter is None:
        max_iter = num_points - 1

    # adjacency / core mask 정리
    clean_adj_list = []
    print("[Server] clean adj_ct start")
    for adj_ct in adjacency_ct_list:
        adj_ct = _refresh(engine, adj_ct, keypack)
        adj_ct = fhe_hard_mask01(engine, adj_ct, num_points, keypack)
        clean_adj_list.append(adj_ct)
    print("[Server] clean adj_ct end")

    core_mask_ct     = _refresh(engine, core_ct, keypack)
    core_mask_ct     = fhe_hard_mask01(engine, core_mask_ct, num_points, keypack)
    non_core_mask_ct = fhe_hard_mask01(
        engine,
        engine.subtract(1.0, core_mask_ct),
        num_points, keypack
    )

    core_labels_ct = _refresh(engine,
        engine.multiply(core_mask_ct, cluster_id_encoded), keypack)
    zero_ct = engine.subtract(core_labels_ct, core_labels_ct)

    # ── 1. Core-Core 전파 ────────────────────────────────
    for iter_idx in range(max_iter):
        print(f"[Server] core-core propagation iteration {iter_idx+1}/{max_iter}")
        for idx, adj_ct in enumerate(clean_adj_list):
            k = idx + 1
            s_labels = _refresh(engine,
                fhe_circular_shift(engine, core_labels_ct, k, num_points, keypack), keypack)
            s_mask   = _refresh(engine,
                fhe_circular_shift(engine, core_mask_ct,   k, num_points, keypack), keypack)

            edge_mask = _refresh(engine,
                engine.multiply(engine.multiply(adj_ct, core_mask_ct, relin_key),
                                 s_mask, relin_key), keypack)

            cand_labels = _refresh(engine,
                engine.multiply(edge_mask, s_labels, relin_key), keypack)

            core_labels_ct = fhe_fast_max_unit_scaled(
                engine, core_labels_ct, cand_labels, num_points, keypack, id_gap=id_gap)
            core_labels_ct = _refresh(engine,
                engine.multiply(core_labels_ct, core_mask_ct, relin_key), keypack)

    # ── 2. Border 할당 ───────────────────────────────────
    border_labels_ct = zero_ct
    assigned_mask_ct = zero_ct

    for idx, adj_ct in enumerate(clean_adj_list):
        print(f"[Server] border assignment {idx+1}/{len(clean_adj_list)}")
        k = idx + 1
        s_labels = _refresh(engine,
            fhe_circular_shift(engine, core_labels_ct, k, num_points, keypack), keypack)
        s_mask   = _refresh(engine,
            fhe_circular_shift(engine, core_mask_ct,   k, num_points, keypack), keypack)

        cand_mask = fhe_hard_mask01(engine,
            _refresh(engine,
                engine.multiply(engine.multiply(adj_ct, s_mask, relin_key),
                                 non_core_mask_ct, relin_key), keypack),
            num_points, keypack)

        empty_mask   = fhe_hard_mask01(engine,
            engine.subtract(1.0, assigned_mask_ct), num_points, keypack)
        accept_mask  = fhe_hard_mask01(engine,
            _refresh(engine,
                engine.multiply(cand_mask, empty_mask, relin_key), keypack),
            num_points, keypack)

        border_labels_ct = _refresh(engine,
            engine.add(border_labels_ct,
                engine.multiply(accept_mask, s_labels, relin_key)), keypack)

        assigned_mask_ct = fhe_hard_mask01(engine,
            fhe_fast_max_unit_scaled(
                engine, assigned_mask_ct, accept_mask,
                num_points, keypack, id_gap=id_gap),
            num_points, keypack)

    final_labels_ct = _refresh(engine,
        engine.add(core_labels_ct, border_labels_ct), keypack)
    return final_labels_ct
